import pandas as pd

import numpy as np

# Load clinical trials dataset

df = pd.read_csv('clinical_outcomes.csv')

# Analyze recovery optimization

baseline = df[df['Treatment'] == 'Control']['Recovery_Days'].mean()

optimized = df[df['Treatment'] == 'Targeted']['Recovery_Days'].mean()

improvement = ((baseline - optimized) / baseline) * 100

print(f"Efficiency Improved By: {improvement:.1f}%")
